# 第13章 平台原生桥接

> 作为跨平台应用，Kazumi 在各平台都有原生代码层。
> 通过 5 个 MethodChannel 桥接各 OS 特有的原生能力：
> 画中画、全屏、外部播放器、桌面快捷方式、存储空间检查等。

### 画中画（PIP）

```kotlin
// 1. 检查 PIP 支持
fun isPictureInPictureSupported(): Boolean {
    return packageManager.hasSystemFeature(PackageManager.FEATURE_PICTURE_IN_PICTURE)
}

// 2. 进入 PIP 模式
fun enterPictureInPicture(): Boolean {
    val params = PictureInPictureParams.Builder()
        .setAspectRatio(Rational(width, height))
        .setAutoEnterEnabled(autoEnterPipOnHomeGesture && pipInPlayerPage)  // Android 12+
        .setActions(buildPipActions())  // 播放/暂停、快进、弹幕开关
        .build()
    return enterPictureInPictureMode(params)
}

// 3. PIP 覆盖层按钮 → BroadcastReceiver → 通知 Flutter
val pipActionReceiver = object : BroadcastReceiver() {
    override fun onReceive(context: Context?, intent: Intent?) {
        when (intent?.action) {
            actionPipPlayPause     → notifyFlutterPipAction("play_pause")
            actionPipForward       → notifyFlutterPipAction("forward")
            actionPipToggleDanmaku → notifyFlutterPipAction("toggle_danmaku")
        }
    }
}

fun notifyFlutterPipAction(action: String) {
    pipChannel?.invokeMethod("onAction", mapOf("action" to action))
}
```

### 全屏（沉浸模式）

```kotlin
fun enterAndroidFullscreen() {
    WindowCompat.setDecorFitsSystemWindows(window, false)
    WindowCompat.getInsetsController(window, window.decorView).apply {
        systemBarsBehavior = BEHAVIOR_SHOW_TRANSIENT_BARS_BY_SWIPE
        hide(WindowInsetsCompat.Type.systemBars())  // 隐藏状态栏 + 导航栏
    }
}

fun exitAndroidFullscreen() {
    WindowCompat.getInsetsController(window, window.decorView)
        .show(WindowInsetsCompat.Type.systemBars())  // 显示系统栏
}
```

### 存储空间

```kotlin
fun getAvailableStorage(path: String): Long {
    val stat = StatFs(path)
    return stat.availableBlocksLong * stat.blockSizeLong
}
```

void FullscreenUtils::EnterNativeFullscreen(HWND hwnd) {
    // 保存原始窗口样式
    DWORD style = GetWindowLongPtr(hwnd, GWL_STYLE);
    // 移除窗口边框
    SetWindowLongPtr(hwnd, GWL_STYLE, style & ~WS_OVERLAPPEDWINDOW);
    // 设置为全屏尺寸
    SetWindowPos(hwnd, HWND_TOP, 0, 0, screenWidth, screenHeight, SWP_FRAMECHANGED);
}

void FullscreenUtils::ExitNativeFullscreen(HWND hwnd) {
    // 恢复原始窗口样式和尺寸
    SetWindowLongPtr(hwnd, GWL_STYLE, originalStyle);
    SetWindowPos(hwnd, NULL, originalRect, SWP_FRAMECHANGED);
}

### 带 Referer 的视频播放

```swift
func openVideoWithReferer(url: String, referer: String) {
    // 弹出选择菜单
    let selectMenu = NSMenu()

    // 选项1：AVPlayer（内置，支持 Referer）
    selectMenu.addItem(NSMenuItem(title: "AVPlayer", action: #selector(openWithAVPlayer)))

    // 选项2：VLC（支持 Referer）
    selectMenu.addItem(NSMenuItem(title: "VLC", action: #selector(openWithVLC)))

    // 选项3：其他应用（不支持 Referer，会提示错误）
    for app in findApplicationsByMimeType() {
        selectMenu.addItem(NSMenuItem(title: appName, action: #selector(openWithSelectedApp)))
    }

    selectMenu.popUp(positioning: nil, at: NSEvent.mouseLocation, in: nil)
}

// AVPlayer：使用 AVURLAssetHTTPHeaderFieldsKey 传递 Referer
@objc func openWithAVPlayer() {
    let headers = ["Referer": httpReferer]
    let asset = AVURLAsset(url: videoUrl!, options: ["AVURLAssetHTTPHeaderFieldsKey": headers])
    let playerItem = AVPlayerItem(asset: asset)
    player = AVPlayer(playerItem: playerItem)
    playerView.player = player
    playerView.player?.play()
}

// VLC：通过命令行参数传递 Referer
@objc func openWithVLC(_ sender: NSMenuItem) {
    let process = Process()
    process.launchPath = "/Applications/VLC.app/Contents/MacOS/VLC"
    process.arguments = [videoUrl!.absoluteString, ":http-referrer=" + httpReferer]
    process.run()
}
```

## 13.9 Dart 侧调用

### 画中画工具

> 源码：`lib/services/player/pip_utils.dart`

```dart
class PipUtils {
    static const pipChannel = MethodChannel('com.predidit.kazumi/pip');

    // 检查 PIP 支持
    static Future<bool> isAndroidPIPSupported() async {
        return await pipChannel.invokeMethod('isPictureInPictureSupported');
    }

    // 进入 PIP
    static Future<bool> enterAndroidPIPWindow({int width = 16, int height = 9}) async {
        return await pipChannel.invokeMethod('enterPictureInPictureMode', {
            'width': width, 'height': height,
        });
    }

    // 更新 PIP 覆盖层按钮
    static Future<void> updateAndroidPIPActions({
        required bool playing, required bool danmakuEnabled,
    }) async {
        await pipChannel.invokeMethod('updatePictureInPictureActions', {
            'playing': playing, 'danmakuEnabled': danmakuEnabled,
        });
    }

    // 接收 PIP 按钮事件（原生→Flutter）
    pipChannel.setMethodCallHandler((call) async {
        if (call.method == 'onAction') {
            final action = call.arguments['action'];
            // 'play_pause' / 'forward' / 'toggle_danmaku'
            switch (action) {
                case 'play_pause': playerController.playOrPause();
                case 'forward': playerController.forward();
                case 'toggle_danmaku': playerController.toggleDanmaku();
            }
        }
    });
}
```

## 13.10 设计模式

| 模式 | 说明 |
|------|------|
| MethodChannel | Flutter ↔ 原生双向通信的标准机制 |
| 平台工厂 | 同一接口，不同平台不同实现（如全屏：Android 沉浸模式 / Win32 无边框） |
| 双向事件 | Flutter→原生：invokeMethod；原生→Flutter：setMethodCallHandler |
| 平台检查 | `Platform.isAndroid` / `Platform.isWindows` 在 Dart 侧分流 |
| 条件注册 | 只在对应平台注册 MethodChannel（如 pip 只在 Android） |

## 13.11 与 Java/Vue 项目的对比

| 概念 | Kazumi (Flutter) | Android 原生 | Web |
|------|------------------|-------------|-----|
| 原生桥接 | MethodChannel | 无需桥接 | 无原生能力 |
| 画中画 | PIP API + MethodChannel | 直接调用 PIP API | PIP API（有限） |
| 全屏 | 沉浸模式 + Win32 API | 直接调用 | Fullscreen API |
| 外部播放器 | ShellExecuteEx / ACTION_VIEW | Intent | 不支持 |
| 菜单栏 | NSMenu + MethodChannel | 无 | 无 |
| 存储检查 | statvfs / StatFs / GetDiskFreeSpaceEx | 直接调用 | navigator.storage |
